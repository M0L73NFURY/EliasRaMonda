Table productos {
  codigo integer [primary key]
  categoria varchar
  nombre varchar
  marca varchar
  id_proveedor integer
  stock_minimo integer
  precio_compra decimal(10,2)
  precio_venta decimal(10,2)
  estado boolean
}

Table inventario {
  id integer [primary key, increment]
  codigo_producto integer
  cantidad integer
  lote varchar
  fecha_vencimiento date
}

Table proveedores {
  id integer [primary key, increment]
  nombre varchar
  rif varchar
  telefono varchar
  email varchar
  direccion text
  activo boolean
}

Table compras {
  id integer [primary key, increment]
  id_proveedor integer
  fecha_compra datetime
  numero_factura varchar
  total decimal(10,2)
  estado varchar // 'pendiente', 'completada', 'cancelada'
}

Table detalle_compras {
  id integer [primary key, increment]
  id_compra integer
  codigo_producto integer
  cantidad integer
  precio_compra decimal(10,2)
  lote varchar
  fecha_vencimiento date
}

Table ventas {
  id integer [primary key, increment]
  fecha_venta datetime
  total decimal(10,2)
  tipo_pago varchar // 'efectivo', 'tarjeta', 'transferencia'
  estado varchar // 'completada', 'cancelada'
}

Table detalle_ventas {
  id integer [primary key, increment]
  id_venta integer
  codigo_producto integer
  cantidad integer
  precio_venta decimal(10,2)
}
